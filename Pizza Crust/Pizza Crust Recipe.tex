	\parbox[t]{0.7\linewidth}{\begin{center}{\textbf{Pizza Crust}}\\\end{center}}
	\end{tabular}}
\begin{center}\noindent\rule{15cm}{0.4pt}\end{center}
\large{Ingredients:}\bigbreak
\normalsize
1.\hspace{0.3cm} $2–2\frac{1}{2}$ cups bread flour.
\smallbreak
2.\hspace{0.3cm}  $1$ Packet yeast
\smallbreak
3.\hspace{0.3cm}  $1\frac{1}{2}$ teaspoons sugar
\smallbreak
4.\hspace{0.3cm} $\frac{3}{4}$ teaspoon salt
\smallbreak
5.\hspace{0.3cm}  $\frac{1}{8}–\frac{1}{4}$ teaspoon garlic powder
\smallbreak
6.\hspace{0.3cm}  $2$ or more tablespoons olive oil
\smallbreak
7.\hspace{0.3cm}  $\frac{3}{4}$ cup warm water
\begin{center}\noindent\rule{15cm}{0.4pt}\end{center}
\large{Preparation:}\bigbreak
\normalsize
1.\hspace{0.3cm}In a large bowl combine: \smallbreak
\hspace{0.6cm}1 cup flour\smallbreak
\hspace{0.6cm}instant yeast\smallbreak
\hspace{0.6cm}sugar\smallbreak
\hspace{0.6cm}salt\smallbreak

2.\hspace{0.3cm}Add olive oil and warm water to large bowl. Mix well.

3.\hspace{0.3cm}Gradually add $1–\frac{1}{3}$ cup of flour. Mix continuously until dough starts to become one continuous ball.

4.\hspace{0.3cm}Completely cover the inside of another large bowl with olive oil.

5.\hspace{0.3cm}Cover your hands in flour then form the dough into a ball. Place the ball into the olive oil covered bowl.

6.\hspace{0.3cm}Let dough rise for 30 minutes.

7.\hspace{0.3cm}After dough has risen, deflate it by gently pressing to the original shape.

8.\hspace{0.3cm}On a lightly floured surface, knead the dough to be smooth.
\break 
\begin{center}\noindent\rule{15cm}{0.4pt}\end{center}
\large{Cooking Instructions:}\bigbreak
\normalsize
1.\hspace{0.3cm}Preheat oven to 425$^\circ$F

2.\hspace{0.3cm}Form dough into 12" circle.

3.\hspace{0.3cm}Fold or pinch edges to create crust.

4.\hspace{0.3cm}Drizzle olive oil(1 tablespoon) over the surface of the crust.

5.\hspace{0.3cm}Poke holes in the center of the crust to prevent bubbles from forming.

6.\hspace{0.3cm}Add toppings and bake in oven for $13–15$ minutes.
